    <script>
        const API_URL = window.location.origin;
        let currentEditId = null;

        // AUTH FAKE BYPASS (Matches AD-102938 / password123 from Neon)
        document.getElementById('login-form').addEventListener('submit', (e) => {
            e.preventDefault();
            document.getElementById('login-section').classList.add('hidden');
            document.getElementById('admin-dashboard').classList.remove('hidden');
            fetchMembers();
        });

        function showTab(t) {
            ['overview', 'members', 'logs'].forEach(tab => {
                document.getElementById('tab-' + tab).classList.add('hidden');
                document.getElementById('btn-' + tab).classList.remove('active-tab', 'text-white');
                document.getElementById('btn-' + tab).classList.add('text-slate-500');
            });
            document.getElementById('tab-' + t).classList.remove('hidden');
            document.getElementById('btn-' + t).classList.add('active-tab', 'text-white');
        }

        function toggleModal(id) { document.getElementById(id).classList.toggle('hidden'); }

        async function fetchMembers() {
            const res = await fetch(`/api/members`);
            const members = await res.json();
            const table = document.getElementById('member-table-body');
            table.innerHTML = '';
            let total = 0;
            members.forEach(m => {
                total += parseFloat(m.balance);
                table.innerHTML += `
                    <tr class="hover:bg-slate-700/20 border-b border-slate-700/50">
                        <td class="p-6 font-bold text-white">${m.full_name}</td>
                        <td class="p-6 font-mono text-blue-400">${m.account_number}</td>
                        <td class="p-6 font-black text-green-400">$${parseFloat(m.balance).toLocaleString()}</td>
                        <td class="p-6 text-center">
                            <button onclick="openEdit(${m.id}, '${m.full_name}', ${m.balance})" class="bg-slate-700 px-5 py-2 rounded-xl text-[10px] font-black hover:bg-white hover:text-black transition">EDIT</button>
                        </td>
                    </tr>`;
            });
            document.getElementById('vault-total').innerText = `$${total.toLocaleString()}`;
            document.getElementById('member-count').innerText = members.length;
        }

        function openEdit(id, name, balance) {
            currentEditId = id;
            document.getElementById('edit-user-name').innerText = name;
            document.getElementById('new-balance-input').value = balance;
            document.getElementById('edit-user-id').value = id;
            toggleModal('edit-balance-modal');
        }

        document.getElementById('register-form').addEventListener('submit', async (e) => {
            e.preventDefault();
            await fetch(`/api/register`, {
                method: 'POST',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    full_name: document.getElementById('reg-name').value,
                    email: document.getElementById('reg-email').value,
                    password: document.getElementById('reg-pass').value
                })
            });
            toggleModal('add-member-modal');
            fetchMembers();
            document.getElementById('register-form').reset();
        });

        document.getElementById('edit-balance-form').addEventListener('submit', async (e) => {
            e.preventDefault();
            await fetch(`/api/update-balance`, {
                method: 'PUT',
                headers: {'Content-Type': 'application/json'},
                body: JSON.stringify({
                    id: currentEditId,
                    balance: document.getElementById('new-balance-input').value
                })
            });
            toggleModal('edit-balance-modal');
            fetchMembers();
        });

        async function fetchLogs() {
            const res = await fetch(`/api/logs`);
            const logs = await res.json();
            document.getElementById('log-table-body').innerHTML = logs.map(l => `
                <tr class="border-b border-slate-700/50">
                    <td class="p-6 text-slate-500 font-mono">${new Date(l.created_at).toLocaleString()}</td>
                    <td class="p-6 font-black text-blue-400">${l.type}</td>
                    <td class="p-6 font-bold text-white">#USR-0${l.user_id}</td>
                    <td class="p-6 font-black text-green-400">$${parseFloat(l.new_balance).toLocaleString()}</td>
                </tr>`).join('');
        }
    </script>
